\subsection{\texorpdfstring{\boldmath}{}Case 2:
    \texorpdfstring{$p < \alpha q$}{p < aq}
    and \texorpdfstring{$p + q \ge 1$}{p + q >= 1}}
\label{sec:case2}

Recall \cref{thm:r,thm:rstar}.
Since $p < \alpha q$ and $p + q \ge 1$, we get
\begin{align*}
r_1^*(q) &= \max\left(\beta + (n-1)\max(0, 1-q), \min(\beta, \alpha q) + (n-1)\min(1, \alpha q)\right),
    \tag{by \cref{thm:rstar}}
\\ r_1(p, q) &= \min(\beta, p) + (n-1)\min(p, 1),
    \tag{by \cref{thm:r}}
\\ r_2^*(p) &= (n-1)\max(1-p, \min(1, p/\alpha)),
    \tag{by \cref{thm:rstar}}
\\ r_2(p, q) &= (n-1)\max(0, 1-p).
    \tag{by \cref{thm:r}, $q \ge 1 - p$}
\end{align*}

\begin{lemma}
\label{thm:2}
If $p < \alpha q$ and $p + q \ge 1$, we get $\mu(p, q) \ge \mu_1$.
\end{lemma}
\begin{proof}
If $p \ge 1$, then $r_2(p, q) = 0$, and if $p = 0$, then $r_1 = 0$.
In both cases, we get $\mu(p, q) = \infty$. So now let $p \in (0, 1)$.

By \cref{thm:p1}, $p_1 \in (\frac{\alpha}{\alpha+1}, 1)$.

\textbf{Case 1}: $p \ge p_1$
\\ Then $p \ge \frac{\alpha}{\alpha+1}$, so $1-p \le p/\alpha$. Thus,
\[ \mu(p, q) \ge \frac{r_2^*(p)}{r_2(p, q)} = \frac{p/\alpha}{1-p}
    \ge \frac{p_1/\alpha}{1-p_1} = \frac{r_2^*(p_1)}{r_2(p_1, q_1)} = \mu_1. \]

\textbf{Case 2}: $p \le p_1$.
\\ By \cref{thm:q1}, $r_1^*(q) \ge r_1^*(q_1)$ for all $q \ge 0$. Thus,
\[ \mu(p, q) \ge \frac{r_1^*(q)}{r_1(p, q)} = \frac{r_1^*(q)}{np}
    \ge \frac{r_1^*(q_1)}{np_1} = \frac{r_1^*(q_1)}{r_1(p_1, q_1)} = \mu_1. \]

Hence, $\mu(p, q) \ge \mu_1$.
\end{proof}
